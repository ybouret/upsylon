Let us assume that we have three different points $\vec{A}$, $\vec{B}$, $\vec{C}$.

\section{Unit Tangent Vectors}

\begin{equation}
\left\lbrace
\begin{array}{rcl}
	\vec{T}_A & = & \myunit{4\vec{B}-3\vec{A}-\vec{C}}\\
	\\
	\vec{T}_B & = & \myunit{\vec{C}-\vec{A}}  \\
	\\
	\vec{T}_C & = & \myunit{3\vec{C}+\vec{A}-4\vec{B}}\\
\end{array}
\right.
\end{equation}
